1use core::result::Result::{Err, Ok};
2
3// ESP32 Hardware abstraction
4use defmt::error;
5use esp_hal::time::{Duration, Rate};
6use esp_hal::{
7    gpio::{AnyPin, Level},
8    rmt::{
9        Channel, ContinuousTxTransaction, LoopMode, PulseCode, Rmt, Tx, TxChannelConfig,
10        TxChannelCreator,
11    },
12    Blocking,
13};
14
15const ESP32S3_RMT_DEFAULT_CLK_FREQ_HZ: u32 = 80_000_000;
16const ESP32S3_RMT_MAX_TX_LOOP_NUM: u32 = 1023; // see ESP32S3 PAC: register is 10 bits wide
17const CLK_FREQ_HZ: u32 = 3_200_000;
18const PULSE_COUNT: usize = 1;
19
20#[derive(Debug)]
21pub enum Error {
22    FreqTooLow(Rate),
23    DurationTooLongForFreq((Duration, Rate)),
24    StopError(esp_hal::rmt::Error),
25    TxError(esp_hal::rmt::Error),
26}
27
28#[derive(Debug)]
29enum Resource<'b> {
30    Channel(Channel<'b, Blocking, Tx>),
31    ContinuousTx(ContinuousTxTransaction<'b>),
32}
33
34/// Buzzer driver using RMT peripheral to generate the signal.
35#[derive(Debug)]
36pub struct Buzzer<'b> {
37    buf: [PulseCode; PULSE_COUNT + 1], // end marker required
38    resource: Resource<'b>,
39}
40
41impl<'b> Buzzer<'b> {
42    /// Build a new buzzer driver.
43    ///
44    /// This requires an RMT peripheral driver `rmt` and the output `pin`
45    pub fn new(rmt: Rmt<'b, Blocking>, pin: AnyPin<'b>) -> Self {
46        let channel = rmt
47            .channel0
48            .configure_tx(
49                &TxChannelConfig::default()
50                    .with_clk_divider(
51                        (ESP32S3_RMT_DEFAULT_CLK_FREQ_HZ / CLK_FREQ_HZ)
52                            .try_into()
53                            .unwrap(),
54                    )
55                    .with_idle_output_level(Level::Low)
56                    .with_idle_output(false)
57                    .with_carrier_modulation(false)
58                    .with_carrier_high(1)
59                    .with_carrier_low(1)
60                    .with_carrier_level(Level::Low),
61            )
62            .expect("RMT channel0 could not configure")
63            .with_pin(pin);
64        Buzzer {
65            buf: [PulseCode::default(), PulseCode::end_marker()],
66            resource: Resource::Channel(channel),
67        }
68    }
69
70    pub fn tone(mut self, freq: Rate, duration: Duration) -> Result<Self, (Self, Error)> {
71        if freq.as_hz() <= CLK_FREQ_HZ / (1 << 16) {
72            // 50% duty on 15 bits is 16 bits for a single PulseCode -> ~49 Hz
73            return Err((self, Error::FreqTooLow(freq)));
74        }
75
76        let wave_length = CLK_FREQ_HZ / (freq.as_hz()); // range checked by freq_hz
77        let count_u32 = (duration.as_millis() as u32 * freq.as_hz()) / 1000;
78        let count = if count_u32 > ESP32S3_RMT_MAX_TX_LOOP_NUM {
79            return Err((self, Error::DurationTooLongForFreq((duration, freq))));
80        } else {
81            count_u32 as u16
82        };
83
84        let high = (wave_length / 2) as u16;
85        let low = (wave_length - high as u32) as u16;
86        self.buf[0] = PulseCode::new(Level::High, high, Level::Low, low);
87
88        let ch = match self.resource {
89            Resource::ContinuousTx(txn) => match txn.stop() {
90                Ok(ch) => ch,
91                Err((e, ch)) => {
92                    return Err((
93                        Buzzer {
94                            buf: self.buf,
95                            resource: Resource::Channel(ch),
96                        },
97                        Error::StopError(e),
98                    ))
99                }
100            },
101            Resource::Channel(ch) => ch,
102        };
103        match ch.transmit_continuously(&self.buf, LoopMode::Finite(count)) {
104            Ok(txn) => Ok(Buzzer {
105                buf: self.buf,
106                resource: Resource::ContinuousTx(txn),
107            }),
108            Err((e, ch)) => {
109                error!("count = {}", count);
110                Err((
111                    Buzzer {
112                        buf: self.buf,
113                        resource: Resource::Channel(ch),
114                    },
115                    Error::TxError(e),
116                ))
117            }
118        }
119    }
120}
